Quick Start

Up and running in minutes

Wrap your app with ThemeProvider, import any component, and start building.

import {
  ThemeProvider,
  ToastProvider,
  Button,
  NovaText,
  Input,
  Card,
  VStack,
  useToast,
} from 'nova-native-ui';

export default function App() {
  return (
    <ThemeProvider darkMode={false}>
      <ToastProvider>
        <VStack spacing="md" padding="lg">
          <NovaText variant="h2">Welcome</NovaText>
          <Card variant="elevated">
            <Input label="Email" placeholder="you@example.com" />
            <Button color="primary">Sign In</Button>
          </Card>
        </VStack>
      </ToastProvider>
    </ThemeProvider>
  );
}
